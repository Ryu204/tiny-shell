107bool launch_executable(const struct args args) {
108 assert(args.argc > 0 && "Invalid args");
109 struct args copied_args;
110 args_deep_copy_init(&copied_args, &args);
111 int pid = fork();
112 int stat_loc = 0;
113 if(pid == -1) {
114 report_error_code(errno);
115 args_destroy(&copied_args);
116 return false;
117 }
118 if(pid == 0) {
119 if(!args.background)
120 signal(SIGINT, SIG_DFL);
121 if(execvp(copied_args.argv[0], copied_args.argv) == -1) {
122 report_error_code(errno);
123 _exit(EXIT_FAILURE);
124 }
125 _exit(EXIT_SUCCESS);
126 } else {
127 if(!args.background) {
128 waitpid(pid, &stat_loc, 0);
129 } else {
130 signal(SIGCHLD, handle_background_child_exit);
131 goto RETURN_TRUE;
132 }
133 if(WIFEXITED(stat_loc)) {
134 const int exit_code = WEXITSTATUS(stat_loc);
135 if(exit_code != 0) {
136 format_success("Exit code: %d\n", exit_code);
137 goto RETURN_FALSE;
138 }
139 goto RETURN_FALSE;
140 } else if(WIFSTOPPED(stat_loc)) {
141 format_error("Stopped\n");
142 goto RETURN_FALSE;
143 } else if(WCOREDUMP(stat_loc)) {
144 format_error("Core dumped\n");
145 goto RETURN_FALSE;
146 } else if(WIFSIGNALED(stat_loc)) {
147 if(WTERMSIG(stat_loc) == SIGSEGV) {
148 format_error("Segmentation fault\n");
149 goto RETURN_FALSE;
150 } else {
151 format_error("Terminated by signal\n");
152 goto RETURN_FALSE;
153 }
154 } else {
155 format_error("Unknown error\n");
156 goto RETURN_FALSE;
157 }
158 RETURN_TRUE:
159 args_destroy(&copied_args);
160 return true;
161 RETURN_FALSE:
162 args_destroy(&copied_args);
163 return false;
164 }
165}

206 for(int i = 0; *(environ + i) != NULL; ++i) {

221 len = new_len;
222 res[new_len - 1] = '\n';
223 }

228bool enum_proc() {
229 DIR *procdir = NULL;
230 struct dirent *entry = NULL;
231 char path[PATH_MAX];
232
233 // Open the /proc directory
234 procdir = opendir("/proc");
235 if(procdir == NULL) {
236 format_error("Cannot open /proc");
237 report_error_code(errno);
238 return false;
239 }
240 // Loop through entries in /proc
241 // NOLINTNEXTLINE
242 while((entry = readdir(procdir)) != NULL) {
243 // Check if entry is a numeric directory (potential process ID)
244 if(!is_number(entry->d_name))
245 continue;
246 // Construct path to process status file
247 snprintf(path, sizeof(path), "/proc/%s/status", entry->d_name);
248 // Open the status file
249 FILE *statusfile = fopen(path, "r");
250 if(statusfile == NULL) {
251 continue;
252 }
253# define NAME_LENGTH 512
254 char name[NAME_LENGTH];
255 // Read the first line for process name
256 if(fscanf(statusfile, "Name: %s512\n", name) != 1) {
257 fclose(statusfile);
258 continue;
259 }
260 // Get PID using entry->d_name (assuming it's numeric)
261 pid_t pid = atoi(entry->d_name);
262 // Get parent process ID
263 char line[NAME_LENGTH];
264 pid_t ppid = -1;
265 // NOLINTBEGIN
266 while(fgets(line, sizeof(line), statusfile) != NULL) {
267 // Find the line starting with "PPid:"
268 if(strncmp(line, "PPid:", 5) == 0) {
269 sscanf(line + 5, "%d", &ppid); // Extract PPID from the line
270 break;
271 }
272 }
273 // NOLINTEND
274 // Get thread count
275 DIR *t_procdir = NULL;
276 char thread_count[3];
277 snprintf(path, sizeof path, "/proc/%d/task", pid);
278 t_procdir = opendir(path);
279
280 if(t_procdir == NULL) {
281 thread_count[0] = '?';
282 thread_count[1] = '\0';
283 } else {
284 unsigned int t_count = 0;
285 struct dirent *entry = NULL;
286 // NOLINTNEXTLINE
287 while((entry = readdir(t_procdir)) != NULL) {
288 if(!is_number(entry->d_name))
289 continue;
290 t_count++;
291 }
292 snprintf(thread_count, sizeof thread_count, "%d", t_count);
293 closedir(t_procdir);
294 }
295 // Print process information
296 printf("PID: %6u PPID: %6u T %3s Name: %s \n", pid, ppid, thread_count, name);
297 fclose(statusfile);
298 }
299 closedir(procdir);
300 return true;
301}

304 enum { buffer_size = 300 };
305 os_char *buffer = malloc(buffer_size * sizeof(os_char));

317
318 mod->argv[0] = buffer;
319
320 mod->argv[1] = malloc(3 * sizeof(os_char));
321 memcpy(mod->argv[1], "-f", 2);
322 mod->argv[1][2] = '\0';
323
324 for(int i = count; i < mod->argc; ++i) {
325 const unsigned int len = strlen(args.argv[i - count]);
326 mod->argv[i] = malloc((len + 1) * sizeof(os_char));
327 memcpy(mod->argv[i], args.argv[i - count], len);
328 mod->argv[i][len] = '\0';
329 }
330 mod->argv[mod->argc] = NULL;
331 const bool res = launch_executable(*mod);
