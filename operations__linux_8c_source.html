68 if(pid == -1) {

73 struct args *copied_args = args_deep_copy(&args);
74 assert(copied_args != NULL);
75 static char *cmds[] = {"e", "a", NULL};
76 if(execvp(copied_args->argv[0], copied_args->argv) == -1) {
77 report_error_code(errno);

79 }
80 _exit(EXIT_SUCCESS);
81 } else {
82 if(!args.background) {
